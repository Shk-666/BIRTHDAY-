    <script>
        // Start the animations and music
        function startExperience() {
            // Hide Overlay
            gsap.to("#overlay", { 
                opacity: 0, 
                duration: 1, 
                onComplete: () => {
                    document.getElementById('overlay').style.display = 'none';
                    playAnimations();
                } 
            });

            // Play Music
            const music = document.getElementById('bgMusic');
            music.play();
            music.volume = 0.4;
        }

        function playAnimations() {
            // Hero Animation
            gsap.to("#hero", { opacity: 1, y: -20, duration: 1.5, ease: "power4.out" });

            // Cards Animation
            gsap.to(".message-card", { 
                opacity: 1, 
                y: 0, 
                duration: 1, 
                stagger: 0.3, 
                delay: 0.5, 
                ease: "back.out(1.7)" 
            });

            // Footer Animation
            gsap.to("#footer", { opacity: 1, duration: 2, delay: 2 });

            // Start Creating Hearts
            setInterval(createHeart, 300);
        }

        // Floating Hearts Logic
        function createHeart() {
            const container = document.getElementById('canvas-container');
            const heart = document.createElement('i');
            heart.className = `fa-solid fa-heart heart`;
            
            // Random styling
            const size = Math.random() * 20 + 10 + 'px';
            const left = Math.random() * 100 + 'vw';
            const duration = Math.random() * 3 + 4 + 's';
            const opacity = Math.random() * 0.5 + 0.3;

            heart.style.left = left;
            heart.style.fontSize = size;
            heart.style.opacity = opacity;
            heart.style.bottom = "-20px";

            container.appendChild(heart);

            // Animate up and fade out
            gsap.to(heart, {
                y: -window.innerHeight - 100,
                x: (Math.random() - 0.5) * 200,
                rotation: Math.random() * 360,
                duration: parseFloat(duration),
                ease: "none",
                onComplete: () => heart.remove()
            });
        }
    </script>
