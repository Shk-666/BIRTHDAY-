    <script>
        // --- Configuration ---
        const heartCount = 30; // Number of hearts on screen
        const music = document.getElementById('bg-music');
        const overlay = document.getElementById('overlay');
        const mainContent = document.getElementById('main-content');
        const musicBtnIcon = document.getElementById('music-icon');
        const musicBtnText = document.getElementById('music-text');
        let isPlaying = true;

        // --- GSAP Animation Setup ---
        
        // Function to create floating hearts
        function createHeart() {
            const heart = document.createElement('div');
            heart.classList.add('fa-solid', 'fa-heart', 'absolute', 'text-love-pink', 'opacity-0', 'glowing-heart');
            
            // Randomize size
            const size = Math.random() * 20 + 10; 
            heart.style.fontSize = `${size}px`;
            
            // Randomize starting position
            heart.style.left = `${Math.random() * 100}%`;
            heart.style.bottom = '-10%';
            
            document.getElementById('heart-container').appendChild(heart);

            // Animate using GSAP
            gsap.to(heart, {
                y: -window.innerHeight - 100,
                x: (Math.random() - 0.5) * 100, // Random wobble
                rotation: Math.random() * 360,
                opacity: Math.random() * 0.5 + 0.2,
                duration: Math.random() * 5 + 5, // 5 to 10 seconds
                ease: 'power1.out',
                onComplete: () => {
                    heart.remove();
                    createHeart(); // Recursively create new hearts
                }
            });
        }

        // --- Interaction Logic ---

        function startExperience() {
            // 1. Play Music
            music.volume = 0.5;
            music.play().catch(e => console.log("Audio requires interaction"));
            
            // 2. Fade out overlay
            gsap.to(overlay, {
                opacity: 0,
                duration: 1,
                onComplete: () => {
                    overlay.style.display = 'none';
                    mainContent.style.display = 'block';
                    mainContent.classList.remove('hidden');
                    
                    // 3. Animate Main Content Entry
                    gsap.to(mainContent, { opacity: 1, duration: 1 });
                    
                    gsap.from(".content-anim", {
                        y: 30,
                        opacity: 0,
                        duration: 1,
                        stagger: 0.2,
                        ease: "back.out(1.7)"
                    });
                }
            });

            // 4. Start Hearts
            for(let i=0; i<heartCount; i++) {
                setTimeout(createHeart, Math.random() * 2000);
            }
        }

        function toggleMusic() {
            if (isPlaying) {
                music.pause();
                musicBtnIcon.classList.remove('fa-pause');
                musicBtnIcon.classList.add('fa-play');
                musicBtnText.innerText = "Play Song";
            } else {
                music.play();
                musicBtnIcon.classList.remove('fa-play');
                musicBtnIcon.classList.add('fa-pause');
                musicBtnText.innerText = "Playing Song";
            }
            isPlaying = !isPlaying;
        }

        // Mouse Move Parallax Effect for Background (Subtle)
        document.addEventListener('mousemove', (e) => {
            const x = (e.clientX / window.innerWidth - 0.5) * 20;
            const y = (e.clientY / window.innerHeight - 0.5) * 20;
            
            gsap.to('#heart-container', {
                x: x,
                y: y,
                duration: 1,
                ease: 'power2.out'
            });
        });
